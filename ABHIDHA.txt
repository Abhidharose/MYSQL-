USE ABHIDHA;

--1.RETRIVE THE DATA

SELECT * FROM CUSTOMERS;

--2.USING WHERE CLAUSE

SELECT * FROM CUSTOMERS WHERE ID =1;

--3.ARRANGING IN ASCENDING OR DESCINDING ORDER IN TABLE WE USE ORDER BY 

SELECT * FROM CUSTOMERS WHERE ID =1 ORDER BY FIRST_NAME;

SELECT * FROM CUSTOMERS ORDER BY FIRST_NAME;

--4.RETRIEVE SPECIFIC COLUMNS

SELECT LAST_NAME,FIRST_NAME FROM CUSTOMERS;

--5.RETRIEVE SALARY ,ALIAS, USING ARITHEMATIC OPERATOR 

SELECT FIRST_NAME,SALARY,(SALARY *10)+100 FROM CUSTOMERS;
SELECT FIRST_NAME,SALARY,(SALARY *10)+100 AS BONOUS FROM CUSTOMERS;

--6.USING DISTINCT KEYWORD TO AVOID SAME DATA IN COLUMNS

SELECT DISTINCT COUNTRY AS UNIQUE_COUNTRY FROM CUSTOMERS; 

--7.USING OF ORDER BY , ARITHEMATIC OPERATORS, ALIAS AND RETRIVING INFORMATION IN CUSTOMER TABLE

SELECT FIRST_NAME,LAST_NAME,SALARY,SALARY*1.1 AS NEW_SALARY FROM CUSTOMERS ORDER BY AGE DESC ; 

--8.RETRIVEING ALL DATA IN THE CUSTOMERS TABLE AND SHOWING NEW SALARY

SELECT *,SALARY * 1.1 AS NEW_SALARY FROM CUSTOMERS;

--9.USING RELATIONAL OPERATOR WITH WHERE CLAUSE

SELECT * FROM CUSTOMERS WHERE AGE>=25;
SELECT * FROM CUSTOMERS WHERE AGE<=25;
SELECT * FROM CUSTOMERS WHERE AGE!=25;
SELECT * FROM CUSTOMERS WHERE AGE=25;

--10. ALTER & UPDATING THE TABLE CUSTOMER USING CASE STATEMENT    

ALTER TABLE CUSTOMERS ADD DOJ DATE;

UPDATE CUSTOMERS SET DOJ=CASE
WHEN CUSTOMER_ID =1 THEN '2020-10-05'
WHEN CUSTOMER_ID =2 THEN '2019-09-17'
WHEN CUSTOMER_ID =3 THEN '2019-12-07'
WHEN CUSTOMER_ID =4 THEN '2023-05-05'
WHEN CUSTOMER_ID =5 THEN '2019-09-17'
ELSE DOJ
 END; 

--11.RENAME TABLE CUSTOMERS TO EMPLOYEES

ALTER TABLE CUSTOMERS RENAME TO EMPLOYEES;

--12.USING LOGICAL OPERATORS AND,OR,NOT
--a.CHECK IF TWO CONDITIONS IN WHERE IS TRUE
SELECT * FROM EMPLOYEES WHERE AGE>22 AND SALARY>=30000; 
--B.CHECK IF ONE CONDITION IS TRUE
SELECT * FROM EMPLOYEES WHERE AGE>22 OR SALARY>=30000; 
--C.USING NOT OPERATOR TO CHECK THE NEGATIVE
SELECT * FROM EMPLOYEES WHERE AGE>22 AND NOT(SALARY>=30000); 

--13.IN & NOT IN OPERATOR
SELECT * FROM EMPLOYEES WHERE CITY  IN('USA','UK');
SELECT * FROM EMPLOYEES WHERE CITY NOT IN('USA','UK'); 

--14.BETWEEN OPERATOR

SELECT FIRST_NAME FROM EMPLOYEES WHERE SALARY BETWEEN 20000 AND 80000;

--15.LIKE OPERATORS
SELECT * FROM EMPLOYEES WHERE LAST_NAME LIKE "B%"; --START WITH B
SELECT * FROM EMPLOYEES WHERE LAST_NAME LIKE "%B"; --END WITH B
SELECT * FROM EMPLOYEES WHERE LAST_NAME LIKE "%B%"; --A 'B' PRESENT ANY WHERE
SELECT * FROM EMPLOYEES WHERE LAST_NAME LIKE "___D";--A 'B' PRESENT EXACTLY AFTER 3 CHARACTERS.
SELECT * FROM EMPLOYEES WHERE FIRST_NAME LIKE 'J%' OR FIRST_NAME LIKE 'D%';

--16.REGEXP OPERATOR(REGULAR EXPRESSION)
SELECT * FROM EMPLOYEES WHERE LAST_NAME REGEXP '^DOE'; --BEGIN WITH DOE 
SELECT * FROM EMPLOYEES WHERE LAST_NAME REGEXP 'DOE$'; --END WITH DOE

--17.HAVING CALUSE AND GROUP BY

select department, count(*) as total_number_employee 
from employees group by department;

select department, avg(salary) as AVG_SALARY 
from employees 
group by department;

select department, avg(salary) as AVG_SALARY  
from employees 
group by department 
having avg(salary) > 40000;

select department, count(*) as total_employee 
from employees 
group by department 
having avg(salary) > 35000 and count(*) > 2;

select department, count(*) as total_employee 
from employees 
where salary > 30000 
group by department 
having count(*) > 2 and avg(salary) > 40000;
